289  template<class KeyT, class ValT>
290  short KDTree<KeyT,ValT>::opt_disc(unsigned kvec_beg,
291  unsigned kvec_end) const
292  {
293  KeyType vmin = init_nodes[kvec_beg].key;
294  KeyType vmax = init_nodes[kvec_beg].key;
295  for(unsigned i=kvec_beg;i<kvec_end;i++)
296  {
297  vmin = CGLA::v_min(vmin,init_nodes[i].key);
298  vmax = CGLA::v_max(vmax,init_nodes[i].key);
299  }
300  short od=0;
301  KeyType ave_v = vmax-vmin;
302  for(unsigned i=1;i<KeyType::get_dim();i++)
303  if(ave_v[i]>ave_v[od]) od = i;
304  return od;
305  }

307  template<class KeyT, class ValT>
308  void KDTree<KeyT,ValT>::optimize(unsigned cur,
309  unsigned kvec_beg,
310  unsigned kvec_end)
311  {
312  // Assert that we are not inserting beyond capacity.
313  assert(cur < nodes.size());
314 
315  // If there is just a single element, we simply insert.
316  if(kvec_beg+1==kvec_end)
317  {
318  nodes[cur] = init_nodes[kvec_beg];
319  nodes[cur].dsc = -1;
320  return;
321  }
322 
323  // Find the axis that best separates the data.
324  short disc = opt_disc(kvec_beg, kvec_end);
325 
326  // Compute the median element. See my document on how to do this
327  // www.imm.dtu.dk/~jab/publications.html
328  unsigned N = kvec_end-kvec_beg;
329  unsigned M = 1<< (CGLA::two_to_what_power(N));
330  unsigned R = N-(M-1);
331  unsigned left_size = (M-2)/2;
332  unsigned right_size = (M-2)/2;
333  if(R < M/2)
334  {
335  left_size += R;
336  }
337  else
338  {
339  left_size += M/2;
340  right_size += R-M/2;
341  }
342 
343  unsigned median = kvec_beg + left_size;
344 
345  // Sort elements but use nth_element (which is cheaper) than
346  // a sorting algorithm. All elements to the left of the median
347  // will be smaller than or equal the median. All elements to the right
348  // will be greater than or equal to the median.
349  const Comp comp(disc);
350  KDNode* data = &init_nodes[0];
351  std::nth_element(data+kvec_beg,
352  data+median,
353  data+kvec_end, comp);
354 
355  // Insert the node in the final data structure.
356  nodes[cur] = init_nodes[median];
357  nodes[cur].dsc = disc;
358 
359  // Recursively build left and right tree.
360  if(left_size>0)
361  optimize(2*cur, kvec_beg, median);
362 
363  if(right_size>0)
364  optimize(2*cur+1, median+1, kvec_end);
365  }

367  template<class KeyT, class ValT>
368  unsigned KDTree<KeyT,ValT>::closest_point_priv(unsigned n, const KeyType& p,
369  ScalarType& dist) const
370  {
371  unsigned ret_node = 0;
372  ScalarType this_dist = nodes[n].dist(p);
373 
374  if(this_dist<dist)
375  {
376  dist = this_dist;
377  ret_node = n;
378  }
379  if(nodes[n].dsc != -1)
380  {
381  short dsc = nodes[n].dsc;
382  ScalarType dsc_dist = CGLA::sqr(nodes[n].key[dsc]-p[dsc]);
383  bool left_son = Comp(dsc)(p,nodes[n].key);
384 
385  if(left_son||dsc_dist<dist)
386  {
387  unsigned left_child = 2*n;
388  if(left_child < nodes.size())
389  if(unsigned nl=closest_point_priv(left_child, p, dist))
390  ret_node = nl;
391  }
392  if(!left_son||dsc_dist<dist)
393  {
394  unsigned right_child = 2*n+1;
395  if(right_child < nodes.size())
396  if(unsigned nr = closest_point_priv(right_child, p, dist))
397  ret_node = nr;
398  }
399  }
400  return ret_node;
401  }

439  NQueue<KDTreeRecord<KeyT, ValT>>& nq) const
440  {
441  ScalarType dist = nodes[n].dist(p);

446  if(nq.at_capacity())
