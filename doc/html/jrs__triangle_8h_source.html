24 /* */
25 /* How to call Triangle from another program */
26 /* */
27 /* */
28 /* If you haven't read Triangle's instructions (run "triangle -h" to read */
29 /* them), you won't understand what follows. */
30 /* */
31 /* Triangle must be compiled into an object file (triangle.o) with the */
32 /* TRILIBRARY symbol defined (generally by using the -DTRILIBRARY compiler */
33 /* switch). The makefile included with Triangle will do this for you if */
34 /* you run "make trilibrary". The resulting object file can be called via */
35 /* the procedure triangulate(). */
36 /* */
37 /* If the size of the object file is important to you, you may wish to */
38 /* generate a reduced version of triangle.o. The REDUCED symbol gets rid */
39 /* of all features that are primarily of research interest. Specifically, */
40 /* the -DREDUCED switch eliminates Triangle's -i, -F, -s, and -C switches. */
41 /* The CDT_ONLY symbol gets rid of all meshing algorithms above and beyond */
42 /* constrained Delaunay triangulation. Specifically, the -DCDT_ONLY switch */
43 /* eliminates Triangle's -r, -q, -a, -u, -D, -Y, -S, and -s switches. */
44 /* */
45 /* IMPORTANT: These definitions (TRILIBRARY, REDUCED, CDT_ONLY) must be */
46 /* made in the makefile or in triangle.c itself. Putting these definitions */
47 /* in this file (triangle.h) will not create the desired effect. */
48 /* */
49 /* */
50 /* The calling convention for triangulate() follows. */
51 /* */
52 /* void triangulate(triswitches, in, out, vorout) */
53 /* char *triswitches; */
54 /* struct triangulateio *in; */
55 /* struct triangulateio *out; */
56 /* struct triangulateio *vorout; */
57 /* */
58 /* `triswitches' is a string containing the command line switches you wish */
59 /* to invoke. No initial dash is required. Some suggestions: */
60 /* */
61 /* - You'll probably find it convenient to use the `z' switch so that */
62 /* points (and other items) are numbered from zero. This simplifies */
63 /* indexing, because the first item of any type always starts at index */
64 /* [0] of the corresponding array, whether that item's number is zero or */
65 /* one. */
66 /* - You'll probably want to use the `Q' (quiet) switch in your final code, */
67 /* but you can take advantage of Triangle's printed output (including the */
68 /* `V' switch) while debugging. */
69 /* - If you are not using the `q', `a', `u', `D', `j', or `s' switches, */
70 /* then the output points will be identical to the input points, except */
71 /* possibly for the boundary markers. If you don't need the boundary */
72 /* markers, you should use the `N' (no nodes output) switch to save */
73 /* memory. (If you do need boundary markers, but need to save memory, a */
74 /* good nasty trick is to set out->pointlist equal to in->pointlist */
75 /* before calling triangulate(), so that Triangle overwrites the input */
76 /* points with identical copies.) */
77 /* - The `I' (no iteration numbers) and `g' (.off file output) switches */
78 /* have no effect when Triangle is compiled with TRILIBRARY defined. */
79 /* */
80 /* `in', `out', and `vorout' are descriptions of the input, the output, */
81 /* and the Voronoi output. If the `v' (Voronoi output) switch is not used, */
82 /* `vorout' may be NULL. `in' and `out' may never be NULL. */
83 /* */
84 /* Certain fields of the input and output structures must be initialized, */
85 /* as described below. */
86 /* */
87 /*****************************************************************************/

89 /*****************************************************************************/
90 /* */
91 /* The `triangulateio' structure. */
92 /* */
93 /* Used to pass data into and out of the triangulate() procedure. */
94 /* */
95 /* */
96 /* Arrays are used to store points, triangles, markers, and so forth. In */
97 /* all cases, the first item in any array is stored starting at index [0]. */
98 /* However, that item is item number `1' unless the `z' switch is used, in */
99 /* which case it is item number `0'. Hence, you may find it easier to */
100 /* index points (and triangles in the neighbor list) if you use the `z' */
101 /* switch. Unless, of course, you're calling Triangle from a Fortran */
102 /* program. */
103 /* */
104 /* Description of fields (except the `numberof' fields, which are obvious): */
105 /* */
106 /* `pointlist': An array of point coordinates. The first point's x */
107 /* coordinate is at index [0] and its y coordinate at index [1], followed */
108 /* by the coordinates of the remaining points. Each point occupies two */
109 /* REALs. */
110 /* `pointattributelist': An array of point attributes. Each point's */
111 /* attributes occupy `numberofpointattributes' REALs. */
112 /* `pointmarkerlist': An array of point markers; one int per point. */
113 /* */
114 /* `trianglelist': An array of triangle corners. The first triangle's */
115 /* first corner is at index [0], followed by its other two corners in */
116 /* counterclockwise order, followed by any other nodes if the triangle */
117 /* represents a nonlinear element. Each triangle occupies */
118 /* `numberofcorners' ints. */
119 /* `triangleattributelist': An array of triangle attributes. Each */
120 /* triangle's attributes occupy `numberoftriangleattributes' REALs. */
121 /* `trianglearealist': An array of triangle area constraints; one REAL per */
122 /* triangle. Input only. */
123 /* `neighborlist': An array of triangle neighbors; three ints per */
124 /* triangle. Output only. */
125 /* */
126 /* `segmentlist': An array of segment endpoints. The first segment's */
127 /* endpoints are at indices [0] and [1], followed by the remaining */
128 /* segments. Two ints per segment. */
129 /* `segmentmarkerlist': An array of segment markers; one int per segment. */
130 /* */
131 /* `holelist': An array of holes. The first hole's x and y coordinates */
132 /* are at indices [0] and [1], followed by the remaining holes. Two */
133 /* REALs per hole. Input only, although the pointer is copied to the */
134 /* output structure for your convenience. */
135 /* */
136 /* `regionlist': An array of regional attributes and area constraints. */
137 /* The first constraint's x and y coordinates are at indices [0] and [1], */
138 /* followed by the regional attribute at index [2], followed by the */
139 /* maximum area at index [3], followed by the remaining area constraints. */
140 /* Four REALs per area constraint. Note that each regional attribute is */
141 /* used only if you select the `A' switch, and each area constraint is */
142 /* used only if you select the `a' switch (with no number following), but */
143 /* omitting one of these switches does not change the memory layout. */
144 /* Input only, although the pointer is copied to the output structure for */
145 /* your convenience. */
146 /* */
147 /* `edgelist': An array of edge endpoints. The first edge's endpoints are */
148 /* at indices [0] and [1], followed by the remaining edges. Two ints per */
149 /* edge. Output only. */
150 /* `edgemarkerlist': An array of edge markers; one int per edge. Output */
151 /* only. */
152 /* `normlist': An array of normal vectors, used for infinite rays in */
153 /* Voronoi diagrams. The first normal vector's x and y magnitudes are */
154 /* at indices [0] and [1], followed by the remaining vectors. For each */
155 /* finite edge in a Voronoi diagram, the normal vector written is the */
156 /* zero vector. Two REALs per edge. Output only. */
157 /* */
158 /* */
159 /* Any input fields that Triangle will examine must be initialized. */
160 /* Furthermore, for each output array that Triangle will write to, you */
161 /* must either provide space by setting the appropriate pointer to point */
162 /* to the space you want the data written to, or you must initialize the */
163 /* pointer to NULL, which tells Triangle to allocate space for the results. */
164 /* The latter option is preferable, because Triangle always knows exactly */
165 /* how much space to allocate. The former option is provided mainly for */
166 /* people who need to call Triangle from Fortran code, though it also makes */
167 /* possible some nasty space-saving tricks, like writing the output to the */
168 /* same arrays as the input. */
169 /* */
170 /* Triangle will not free() any input or output arrays, including those it */
171 /* allocates itself; that's up to you. You should free arrays allocated by */
172 /* Triangle by calling the trifree() procedure defined below. (By default, */
173 /* trifree() just calls the standard free() library procedure, but */
174 /* applications that call triangulate() may replace trimalloc() and */
175 /* trifree() in triangle.c to use specialized memory allocators.) */
176 /* */
177 /* Here's a guide to help you decide which fields you must initialize */
178 /* before you call triangulate(). */
179 /* */
180 /* `in': */
181 /* */
182 /* - `pointlist' must always point to a list of points; `numberofpoints' */
183 /* and `numberofpointattributes' must be properly set. */
184 /* `pointmarkerlist' must either be set to NULL (in which case all */
185 /* markers default to zero), or must point to a list of markers. If */
186 /* `numberofpointattributes' is not zero, `pointattributelist' must */
187 /* point to a list of point attributes. */
188 /* - If the `r' switch is used, `trianglelist' must point to a list of */
189 /* triangles, and `numberoftriangles', `numberofcorners', and */
190 /* `numberoftriangleattributes' must be properly set. If */
191 /* `numberoftriangleattributes' is not zero, `triangleattributelist' */
192 /* must point to a list of triangle attributes. If the `a' switch is */
193 /* used (with no number following), `trianglearealist' must point to a */
194 /* list of triangle area constraints. `neighborlist' may be ignored. */
195 /* - If the `p' switch is used, `segmentlist' must point to a list of */
196 /* segments, `numberofsegments' must be properly set, and */
197 /* `segmentmarkerlist' must either be set to NULL (in which case all */
198 /* markers default to zero), or must point to a list of markers. */
199 /* - If the `p' switch is used without the `r' switch, then */
200 /* `numberofholes' and `numberofregions' must be properly set. If */
201 /* `numberofholes' is not zero, `holelist' must point to a list of */
202 /* holes. If `numberofregions' is not zero, `regionlist' must point to */
203 /* a list of region constraints. */
204 /* - If the `p' switch is used, `holelist', `numberofholes', */
205 /* `regionlist', and `numberofregions' is copied to `out'. (You can */
206 /* nonetheless get away with not initializing them if the `r' switch is */
207 /* used.) */
208 /* - `edgelist', `edgemarkerlist', `normlist', and `numberofedges' may be */
209 /* ignored. */
210 /* */
211 /* `out': */
212 /* */
213 /* - `pointlist' must be initialized (NULL or pointing to memory) unless */
214 /* the `N' switch is used. `pointmarkerlist' must be initialized */
215 /* unless the `N' or `B' switch is used. If `N' is not used and */
216 /* `in->numberofpointattributes' is not zero, `pointattributelist' must */
217 /* be initialized. */
218 /* - `trianglelist' must be initialized unless the `E' switch is used. */
219 /* `neighborlist' must be initialized if the `n' switch is used. If */
220 /* the `E' switch is not used and (`in->numberofelementattributes' is */
221 /* not zero or the `A' switch is used), `elementattributelist' must be */
222 /* initialized. `trianglearealist' may be ignored. */
223 /* - `segmentlist' must be initialized if the `p' or `c' switch is used, */
224 /* and the `P' switch is not used. `segmentmarkerlist' must also be */
225 /* initialized under these circumstances unless the `B' switch is used. */
226 /* - `edgelist' must be initialized if the `e' switch is used. */
227 /* `edgemarkerlist' must be initialized if the `e' switch is used and */
228 /* the `B' switch is not. */
229 /* - `holelist', `regionlist', `normlist', and all scalars may be ignored.*/
230 /* */
231 /* `vorout' (only needed if `v' switch is used): */
232 /* */
233 /* - `pointlist' must be initialized. If `in->numberofpointattributes' */
234 /* is not zero, `pointattributelist' must be initialized. */
235 /* `pointmarkerlist' may be ignored. */
236 /* - `edgelist' and `normlist' must both be initialized. */
237 /* `edgemarkerlist' may be ignored. */
238 /* - Everything else may be ignored. */
239 /* */
240 /* After a call to triangulate(), the valid fields of `out' and `vorout' */
241 /* will depend, in an obvious way, on the choice of switches used. Note */
242 /* that when the `p' switch is used, the pointers `holelist' and */
243 /* `regionlist' are copied from `in' to `out', but no new space is */
244 /* allocated; be careful that you don't free() the same array twice. On */
245 /* the other hand, Triangle will never copy the `pointlist' pointer (or any */
246 /* others); new space is allocated for `out->pointlist', or if the `N' */
247 /* switch is used, `out->pointlist' remains uninitialized. */
248 /* */
249 /* All of the meaningful `numberof' fields will be properly set; for */
250 /* instance, `numberofedges' will represent the number of edges in the */
251 /* triangulation whether or not the edges were written. If segments are */
252 /* not used, `numberofsegments' will indicate the number of boundary edges. */
253 /* */
254 /*****************************************************************************/
