  1from pygel3d import hmesh, lib_py_gel, IntVector
  2import ctypes as ct
  3import numpy as np

  5class Graph:
  6    """ This class is for representing graphs embedded in 3D. The class does not in
  7    itself come with many features: it contains methods for creating, accessing, and
  8    housekeeping. When vertices are used as parameters in the functions below, we usually
  9    use the parameter name n (for node). n is simply an index (i.e. an integer) that
 10    refers to a node (aka vertex)."""
 11    def __init__(self,orig=None):
 12        if orig == None:
 13            self.obj = lib_py_gel.Graph_new()
 14        else:
 15            self.obj = lib_py_gel.Graph_copy(orig.obj)
 16    def __del__(self):
 17        lib_py_gel.Graph_delete(self.obj)
 18    def clear(self):
 19        """ Clear the graph. """
 20        lib_py_gel.Graph_clear(self.obj)
 21    def cleanup(self):
 22        """ Cleanup reorders the graph nodes such that there is no
 23        gap in the index range. """
 24        lib_py_gel.Graph_cleanup(self.obj)
 25    def nodes(self):
 26        """ Get all nodes as an iterable range """
 27        nodes = IntVector()
 28        lib_py_gel.Graph_nodes(self.obj, nodes.obj)
 29        return nodes
 30    def neighbors(self, n, mode='n'):
 31        """ Get the neighbors of node n. The final argument is either 'n' or 'e'. If it is 'n'
 32        the function returns all neighboring nodes, and if it is 'e' it returns incident edges."""
 33        nbors = IntVector()
 34        lib_py_gel.Graph_neighbors(self.obj, n, nbors.obj, ct.c_char(mode.encode('ascii')))
 35        return nbors
 36    def positions(self):
 37        """ Get the vertex positions by reference. You can assign to the
 38        positions. """
 39        pos = ct.POINTER(ct.c_double)()
 40        n = lib_py_gel.Graph_positions(self.obj, ct.byref(pos))
 41        return np.ctypeslib.as_array(pos,(n,3))
 42    def average_edge_length(self):
 43        """ Returns the average edge length. """
 44        ael = lib_py_gel.Graph_average_edge_length(self.obj)
 45        return ael
 46    def add_node(self, p):
 47        """ Adds node with position p to the graph and returns the
 48        index of the new node. """
 49        return lib_py_gel.Graph_add_node(self.obj, np.array(p))
 50    def remove_node(self, n):
 51        """ Removes the node n passed as argument. This does not change
 52        any indices of other nodes, but n is then invalid. """
 53        lib_py_gel.Graph_remove_node(self.obj, n)
 54    def node_in_use(self, n):
 55        """ Checks if n is in_use. This function returns false both
 56        if n has been removed and if n is an index outside the range of
 57        indices that are used. """
 58        return lib_py_gel.Graph_node_in_use(self.obj, n)
 59    def connect_nodes(self, n0, n1):
 60        """ Creates a new edge connecting nodes n0 and n1. The index of
 61        the new edge is returned. """
 62        return lib_py_gel.Graph_connect_nodes(self.obj, n0, n1)
 63    def disconnect_nodes(self, n0, n1):
 64        """ Disconect nodes n0 and n1"""
 65        lib_py_gel.Graph_disconnect_nodes(self.obj, n0, n1)
 66    def merge_nodes(self, n0, n1, avg_pos):
 67        """ Merge nodes n0 and n1. avg_pos indicates if you want the position to be the average. """
 68        lib_py_gel.Graph_merge_nodes(self.obj, n0, n1, avg_pos)

 71def from_mesh(m):
 72    """ Creates a graph from a mesh. The argument, m, is the input mesh,
 73    and the function returns a graph with the same vertices and edges
 74    as m."""
 75    g = Graph()
 76    lib_py_gel.graph_from_mesh(m.obj, g.obj)

 79def load(fn):
 80    """ Load a graph from a file. The argument, fn, is the filename which
 81    is in a special format similar to Wavefront obj. The loaded graph is
 82    returned by the function - or None if loading failed. """
 83    s = ct.c_char_p(fn.encode('utf-8'))
 84    g = Graph()
 85    if lib_py_gel.graph_load(g.obj, s):
 86        return g
 87    return None

 89def save(fn, g):
 90    """ Save graph to a file. The first argument, fn, is the file name,
 91    and g is the graph. This function returns True if saving happened and
 92    False otherwise. """
 93    s = ct.c_char_p(fn.encode('utf-8'))
 94    return lib_py_gel.graph_save(g.obj, s)

 96def to_mesh_cyl(g, fudge):
 97    """ Creates a Manifold mesh from the graph. The first argument, g, is the
 98    graph we want converted, and fudge is a constant that is used to increase the radius
 99    of every node. This is useful if the radii are 0. """
100    m = hmesh.Manifold()
101    lib_py_gel.graph_to_mesh_cyl(g.obj, m.obj, fudge)
102    return m

104def smooth(g, iter=1, alpha=1.0):
105    """ Simple Laplacian smoothing of a graph. The first argument is the Graph, g, iter
106    is the number of iterations, and alpha is the weight. If the weight is high,
107    each iteration causes a lot of smoothing, and a high number of iterations
108    ensures that the effect of smoothing diffuses throughout the graph, i.e. that the
109    effect is more global than local. """
110    lib_py_gel.graph_smooth(g.obj, iter, alpha)

112def edge_contract(g, dist_thresh):
113    """ Simplifies a graph by contracting edges. The first argument, g, is the graph,
114    and only edges shorter than dist_thresh are contracted. When an edge is contracted
115    the merged vertices are moved to the average of their former positions. Thus,
116    the ordering in which contractions are carried out matters. Hence, edges are
117    contracted in the order of increasing length and edges are only considered if
118    neither end point is the result of a contraction, but the process is then repeated
119    until no more contractions are possible. Returns total number of contractions. """
120    return lib_py_gel.graph_edge_contract(g.obj, dist_thresh)

122def prune(g):
123    """ Prune leaves of a graph. The graph, g, is passed as the argument. This function
124        removes leaf nodes (valency 1) whose only neighbour has valency > 2. In practice
125        such isolated leaves are frequently spurious if the graph is a skeleton. Does not
126        return a value. """
127    lib_py_gel.graph_prune(g.obj)

129def LS_skeleton(g, sampling=True):
130    """ Skeletonize a graph using the local separators approach. The first argument,
131        g, is the graph, and, sampling indicates whether we try to use all vertices
132        (False) as starting points for finding separators or just a sampling (True).
133        The function returns a new graph which is the skeleton of the input graph. """
134    skel = Graph()
135    mapping = IntVector()
136    lib_py_gel.graph_LS_skeleton(g.obj, skel.obj, mapping.obj, sampling)
137    return skel

139def LS_skeleton_and_map(g, sampling=True):
140    """ Skeletonize a graph using the local separators approach. The first argument,
141        g, is the graph, and, sampling indicates whether we try to use all vertices
142        (False) as starting points for finding separators or just a sampling (True).
143        The function returns a tuple containing a new graph which is the skeleton of
144        the input graph and a map from the graph nodes to the skeletal nodes. """

146    mapping = IntVector()
147    lib_py_gel.graph_LS_skeleton(g.obj, skel.obj, mapping.obj, sampling)
148    return skel, mapping

150def front_skeleton_and_map(g, colors):
151    """ Skeletonize a graph using the front separators approach. The first argument,
152        g, is the graph, and, colors is a 2D array where each row contains a sequence
153        of floating point values - one for each node. We can have as many rows as needed
154        for the front separator computation. We can think of this as a coloring
155        of the nodes, hence the name. In practice, a coloring might just be the x-coordinate
156        of the nodes or some other function that indicates something about the structure of the
157        graph. The function returns a tuple containing a new graph which is the
158        skeleton of the input graph and a map from the graph nodes to the skeletal nodes. """
159    skel = Graph()
160    mapping = IntVector()
161    colors_flat = np.asarray(colors, dtype=np.float64)
162    N_col = colors_flat.shape[0]
163    lib_py_gel.graph_front_skeleton(g.obj, skel.obj, mapping.obj, N_col, colors_flat.ctypes.data_as(ct.POINTER(ct.c_double)))
164    return skel, mapping
